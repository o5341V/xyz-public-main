// Calculating High Z Score

z_score(_window) =>
    a = ta.sma(high, _window)
    b = ta.stdev(high, _window)
    c = (high - a) / b 
    d = ta.sma(low, _window)
    e = ta.stdev(low, _window)
    f = (low - d) / e 
    z = (c + f) / 2
